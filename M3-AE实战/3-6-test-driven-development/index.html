    test-driven-development¶

    这个 skill 做什么¶

    test-driven-development (TDD) 是一个写代码的纪律——先写一个会失败的测试，再写刚好够让它通过的代码，然后重构。在嵌入式场景下，TDD 的难点是硬件依赖——你得区分"这个测试失败了是因为代码逻辑错了"还是"因为硬件没接"。这个 skill 帮你做这个区分。

    类比：TDD 是修路先放路障——在你要修的路段前面放一个"此路不通，请绕行"的牌子（失败测试），等你修好路了（实现代码），路障撤掉（测试通过）。你永远不会"以为路修好了但实际上还有个大坑"。

    什么时候用 / 什么时候不该用¶

    ✅ 应该用¶

    场景 为什么需要 TDD
    写一个新的驱动或模块 测试先行确保每个函数的行为被验证
    修复一个逻辑类 bug（没有硬件不确定性） 先写一个失败测试重现 bug → 修复 → 测试通过 → bug 确认修复
    涉及状态转换的逻辑（如通信协议状态机） 状态转换错误很难在集成测试中发现——单元测试能精确覆盖每个转换
    数据处理函数（CRC 校验、数据打包/解包、滤波算法） 纯逻辑，无硬件依赖——TDD 最有价值的场景

    ❌ 不该用¶

    场景 为什么不该用 替代方案
    纯硬件初始化代码（MX_GPIO_Init、时钟配置） 这些代码是 CubeMX 生成的，行为依赖硬件寄存器——测试的前提不成立 依赖集成测试（在真实硬件上跑）
    紧急 bug fix 需要 5 分钟内上线 TDD 需要时间写测试——紧急时不可行 先修 bug，事后补测试
    探索性原型（不确定怎么设计最合适） TDD 需要你先知道"期待的行为是什么"——原型阶段你也不确定 先 incremental-implementation，稳定后再补测试

    怎么触发¶

    触发方式¶

    用 TDD 给这个 GPIO 输出模块写单元测试和实现

    或更精确：

    我要给 i2c_eeprom.c 中的 EEPROM_Write 函数补一个"长度校验"的输入验证——先帮我写一个测试，确认长度超限时返回 -1

    TDD 循环¶

    RED（红）    →  写一个失败测试
GREEN（绿）  →  写最少的代码让测试通过
REFACTOR     →  改进代码结构，测试仍然通过

    AI 会自动引导你完成这三个步骤。

    嵌入式 TDD 实例：GPIO 输出抽象¶

    场景：你想封装一个 GPIO 输出操作，屏蔽 HAL 细节——业务代码不直接调 HAL_GPIO_WritePin，而是通过你的抽象层。现在用 TDD 来写这个抽象层。

    Step 1: RED — 写一个失败测试¶

    你（或 AI）先写测试：

    /* test_gpio_output.c — Unity test framework */

#include "unity.h"
#include "mock_hal.h"           /* 模拟 HAL 调用 */
#include "bsp/gpio_output.h"
#include <stdbool.h>

void setUp(void) { mock_hal_reset(); }
void tearDown(void) {}

/* Test 1: 初始化有效引脚应返回成功 */
void test_gpio_init_valid_pin_returns_ok(void) {
    gpio_status_t status = gpio_output_init(PIN_LED_GREEN, GPIO_MODE_OUTPUT_PP);
    TEST_ASSERT_EQUAL(GPIO_OK, status);
}

/* Test 2: 初始化超出范围的引脚应返回错误 */
void test_gpio_init_out_of_range_pin_returns_error(void) {
    gpio_status_t status = gpio_output_init(GPIO_PIN_COUNT + 1, GPIO_MODE_OUTPUT_PP);
    TEST_ASSERT_EQUAL(GPIO_ERROR_INVALID_PIN, status);
}

/* Test 3: 未初始化的引脚写操作应返回错误（安全约束） */
void test_gpio_write_without_init_returns_error(void) {
    gpio_status_t status = gpio_output_write(PIN_LED_GREEN, GPIO_LEVEL_HIGH);
    TEST_ASSERT_EQUAL(GPIO_ERROR_NOT_INITIALIZED, status);
}

/* Test 4: 写入 HIGH 时应调用 HAL_GPIO_WritePin 设置引脚 */
void test_gpio_write_high_calls_hal_correctly(void) {
    gpio_output_init(PIN_LED_GREEN, GPIO_MODE_OUTPUT_PP);
    gpio_status_t status = gpio_output_write(PIN_LED_GREEN, GPIO_LEVEL_HIGH);
    TEST_ASSERT_EQUAL(GPIO_OK, status);
    /* mock 记录 HAL 被调用时的参数 */
    TEST_ASSERT_EQUAL_PTR(GPIOA, mock_hal_get_last_port());
    TEST_ASSERT_EQUAL(GPIO_PIN_5, mock_hal_get_last_pin());
    TEST_ASSERT_EQUAL(GPIO_PIN_SET, mock_hal_get_last_state());
}

    ⚠️ 注意：Test 4 直接验证 HAL 调用参数——这属于"实现验证"（white-box test）。在真正的 TDD 中，更偏好行为验证（"引脚状态变成了 HIGH"而不是"HAL 函数 X 被调用"）。这里为了教学展示 mock 机制，保留了 white-box 风格；实际项目中优先行为验证。

    运行测试 → 全红（函数还没实现）✅ — 这就是 RED 状态。

    Step 2: GREEN — 实现刚好够让测试通过¶

    AI 生成最小实现：

    /* bsp/gpio_output.h */
#ifndef BSP_GPIO_OUTPUT_H
#define BSP_GPIO_OUTPUT_H

#include <stdint.h>
#include <stdbool.h>

/* 引脚索引——通过查找表映射到实际的 GPIO 端口+引脚 */
#define GPIO_PIN_COUNT   16
#define PIN_LED_GREEN    0    /* → GPIOA, GPIO_PIN_5 */
#define PIN_LED_RED      1    /* → GPIOA, GPIO_PIN_6 */

/* 抽象层的模式和电平枚举——隔离 HAL 的具体常量 */
typedef enum {
    GPIO_MODE_OUTPUT_PP = 0,   /* 推挽输出——映射到 HAL: GPIO_MODE_OUTPUT_PP */
} gpio_output_mode_t;

typedef enum {
    GPIO_LEVEL_LOW = 0,
    GPIO_LEVEL_HIGH = 1,
} gpio_level_t;

typedef enum {
    GPIO_OK = 0,
    GPIO_ERROR_INVALID_PIN,
    GPIO_ERROR_NOT_INITIALIZED,
} gpio_status_t;

gpio_status_t gpio_output_init(uint8_t pin_index, gpio_output_mode_t mode);
gpio_status_t gpio_output_write(uint8_t pin_index, gpio_level_t level);

#endif /* BSP_GPIO_OUTPUT_H */

    /* bsp/gpio_output.c */
#include "bsp/gpio_output.h"
#include "stm32l4xx_hal.h"
#include <stdbool.h>

/* 引脚查找表：pin_index → {GPIO_TypeDef*, pin_number} */
typedef struct {
    GPIO_TypeDef *port;
    uint16_t      pin;
    bool          initialized;
} gpio_pin_config_t;

static gpio_pin_config_t pin_config[GPIO_PIN_COUNT] = {
    [PIN_LED_GREEN] = {GPIOA, GPIO_PIN_5, false},
    [PIN_LED_RED]   = {GPIOA, GPIO_PIN_6, false},
    /* ... 其余引脚在 CubeMX 初始化时填充 */
};

static bool is_valid_pin(uint8_t pin_index) {
    return pin_index < GPIO_PIN_COUNT && pin_config[pin_index].port != NULL;
}

gpio_status_t gpio_output_init(uint8_t pin_index, gpio_output_mode_t mode) {
    if (!is_valid_pin(pin_index)) return GPIO_ERROR_INVALID_PIN;

    GPIO_InitTypeDef gpio_init = {
        .Pin   = pin_config[pin_index].pin,
        .Mode  = mode,               /* 抽象 mode 直接对应 HAL mode */
        .Pull  = GPIO_NOPULL,
        .Speed = GPIO_SPEED_FREQ_LOW,
    };
    HAL_GPIO_Init(pin_config[pin_index].port, &gpio_init);
    pin_config[pin_index].initialized = true;
    return GPIO_OK;
}

gpio_status_t gpio_output_write(uint8_t pin_index, gpio_level_t level) {
    if (!is_valid_pin(pin_index)) return GPIO_ERROR_INVALID_PIN;
    if (!pin_config[pin_index].initialized) return GPIO_ERROR_NOT_INITIALIZED;

    GPIO_PinState hal_state = (level == GPIO_LEVEL_HIGH) ? GPIO_PIN_SET : GPIO_PIN_RESET;
    HAL_GPIO_WritePin(pin_config[pin_index].port, pin_config[pin_index].pin, hal_state);
    return GPIO_OK;
}

    运行测试 → 全绿 ✅ — GREEN 状态。

    Step 3: REFACTOR — 改进结构¶

    测试通过后，AI 建议改进代码结构（保持测试通过）：

    AI: 注意到 HAL_GPIO_Init 的配置结构体 gpio_init 在 gpio_output_init 中
    每次调用都要初始化。可以提取为辅助函数：

    static void gpio_config_pin(gpio_pin_config_t *cfg, gpio_output_mode_t mode) {
        GPIO_InitTypeDef init = {
            .Pin = cfg->pin, .Mode = mode,
            .Pull = GPIO_NOPULL, .Speed = GPIO_SPEED_FREQ_LOW
        };
        HAL_GPIO_Init(cfg->port, &init);
    }

    另外，is_valid_pin() 已经在 gpio_output_init 和 gpio_output_write 中复用——
    边界检查逻辑没有重复。但 pin_config 数组的 initialized 字段可以在
    HAL_GPIO_DeInit 被调用时重置，为将来支持引脚释放做准备。

    运行测试 → 仍然全绿 ✅ — REFACTOR 成功。

    嵌入式测试的特殊挑战¶

    挑战 1：硬件依赖¶

    问题：HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5, GPIO_PIN_SET) 在没有硬件时无法执行。

    方案（按推荐优先级）：

    方案 适用场景 复杂性
    Mock（模拟 HAL 调用） 单元测试（在 PC 上跑）——验证逻辑，不验证硬件行为 低
    接口抽象（用函数指针替换 HAL） 驱动层测试——可以在测试中注入虚拟硬件行为 中
    HIL 测试（Hardware-in-Loop） 集成测试——在真实 MCU 上运行，验证硬件行为 高（需要硬件+烧录）

    Mock 示例（PC 上跑，不连硬件）：

    /* mock_hal.h — 拦截所有 HAL 调用 */
#define HAL_GPIO_WritePin(port, pin, state) \
    mock_hal_record_gpio_write(port, pin, state)

    💡 关键原则：单元测试测逻辑（用 mock），集成测试测硬件（用 HIL）。不要试图在单元测试中测硬件行为。

    挑战 2：中断和定时器¶

    中断处理函数的行为很难在传统单元测试中覆盖。策略：

    • 中断逻辑尽量薄：ISR 中只做标志位记录（如 i2c_event_occurred = true），实际处理放在主循环
    • 主循环逻辑用 TDD：可以单元测试的部分拉出来
    • ISR 本身用集成测试：在真实硬件上用逻辑分析仪或 GPIO toggle 验证

    挑战 3：测试框架选型¶

    框架 适合 说明
    Unity 裸机 / FreeRTOS 项目 轻量（3 个文件）、纯 C、嵌入式专用
    CppUTest 需要 mock 的复杂项目 C++ 编译但测试 C 代码，Mock 能力强
    Ceedling 需要完整测试工具链 封装了 Unity + CMock + CException，一键生成 mock

    对于嵌入式 TDD 入门，建议从 Unity 开始。

    做对了的标准是什么¶

    ✅ 成功输出¶

    标准 说明
    测试先红后绿 测试在实现前写——先看到失败，实现后看到通过
    测试覆盖边界条件 不只是 happy path——包含无效输入、边界值、未初始化状态
    每个测试验证一个行为 测试命名描述具体行为（test_gpio_write_without_init_returns_error），不是"test_gpio"
    实现刚好够通过 没有多余的"未来可能需要"的代码

    ❌ 失败输出¶

    失败表现 为什么不行 怎么办
    实现先于测试（或者实现和测试"同时"写） 这不是 TDD——没有 RED 步骤，你可能写了"测试为了通过实现"而不是"实现为了通过测试" 删掉实现，先写测试，确认测试失败
    测试只覆盖 happy path 这样的测试给人虚假的安全感——代码在异常输入下行为未知 每个函数至少加 1 个异常输入测试 + 1 个边界值测试
    测试依赖硬件，没有 mock 测试只能在连了硬件的机器上跑——CI 里跑不了，每次改代码都要烧录验证 用 mock 分离逻辑和硬件依赖

    常见误区¶

    误区 1："嵌入式代码测不了，都得连硬件"¶

    现象：工程师说"我的代码要操作寄存器，单元测试没意义"。

    正确做法：分离逻辑和硬件——数据处理、状态机、协议解析、输入验证都是纯逻辑，完全可以单元测试。GPIO/SPI/I2C 的寄存器操作是这个逻辑的"叶子节点"——通过接口抽象或 mock 替换。

    判断标准：如果你的函数里有 if/else/switch/while，它包含了逻辑——这个逻辑值得测试。只有纯寄存器赋值（GPIOA->BSRR = 0x0020）才不需要单元测试。

    误区 2："TDD 太慢了，直接写代码更快"¶

    现象：你觉得写测试耽误时间——"功能我已经想清楚了，直接实现就行"。

    正确做法：TDD 的时间不是"额外"的——它取代了"写完代码后手动验证"的时间。对于一个带几个边界条件的中等函数（如数据校验），手动验证 = 烧录 + 构造各种输入 + 串口打印观察 = 10-20 分钟。TDD = 写 3-4 个测试（含首次 mock 搭建 ~15 分钟，后续复用）+ 运行 = 明显更短。首次 TDD 设置（Unity 框架、mock 基础设施、构建配置）需要 30-60 分钟，但这是一次性投入——后续每个函数的 TDD 只需 5-15 分钟。

    误区 3："代码改动后测试也要改，太麻烦"¶

    现象：你改了代码逻辑，测试失败了——你觉得得花时间改测试，"不如不写测试"。

    正确做法：测试失败 = 测试在履行它的职责。它告诉你了"代码行为变了"——这正是测试的价值。如果你改代码时测试不失败，说明测试没覆盖到你改的路径。

    进阶¶

    TDD + incremental-implementation¶

    TDD 和 incremental-implementation 天然配对——每个垂直切片 = 一组测试 + 实现。下一节详讲。

    Mock 策略深入¶

    对于更复杂的 mock（如模拟 I2C 从机行为、注入通信错误），工具链建议：

    Unity + CMock → Ceedling 一键生成 mock 函数

    测试覆盖率¶

    当测试积累到一定数量后，用覆盖率工具（gcov/lcov）检查哪些代码路径未被测试覆盖：

    # 编译时加覆盖率标志（Unity 通常以源文件形式编译，不使用 -lunity）
gcc -fprofile-arcs -ftest-coverage -o test test_gpio.c gpio_output.c unity/unity.c
./test
gcov gpio_output.c  # 查看覆盖率报告
